include_directories (${FLAKE_INCLUDES} )

    WebShape.cpp
    WebShapeFactory.cpp
    WebShapePlugin.cpp

    WebToolFactory.cpp
)

    ${QT_QTCORE_LIBRARY}
    ${QT_QTWEBKIT_LIBRARY}
